rb-cm-client.maude

         cp2-interfaces 
  mod CONTENT-MANAGER-CLIENT incs CONTENT-MANAGER 
        +MARKOV-ACTION-MODEL+CP2-COMMON+MC-INTERFACE
        +ED-INTERFACE+TGEN-INTERFACE 
  mod CM-CLIENT-TEST incs CONTENT-MANAGER-CLIENT
